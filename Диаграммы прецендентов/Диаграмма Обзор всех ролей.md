```mermaid
flowchart LR
    subgraph Customers["Клиенты"]
        A1["Покупатель"]
        A2["Продавец"]
    end
    
    subgraph Operations["Операции"]
        A3["Склад"]
        A4["Курьер"]
        A5["Модератор"]
        A6["Категории"]
    end
    
    subgraph Support["Поддержка"]
        A7["Поддержка"]
        A8["Менеджер продавцов"]
    end
    
    subgraph Management["Управление"]
        A9["Финансы"]
        A10["Маркетинг"]
        A11["Аналитик"]
        A14["Администратор"]
    end
    
    subgraph Tech["Технические"]
        A12["Системный админ"]
        A13["Аудитор"]
    end
    
    %% Ключевые взаимодействия
    A1 -.->|заказы| A3
    A1 -.->|справка| A7
    A1 -.->|споры| A14
    
    A2 -.->|товары| A3
    A2 -.->|вопросы| A8
    A2 -.->|справка| A7
    
    A3 -.->|передача| A4
    A3 -.->|модерация| A5
    
    A5 -.->|категории| A6
    
    A7 -.->|эскалация| A8
    A7 -.->|споры| A14
    
    A8 -.->|управление| A2
    
    A9 -.->|отчеты| A11
    A9 -.->|аудит| A13
    
    A10 -.->|акции| A14
    
    A14 -.->|управление| A12
    
    classDef customer fill:#e8f5e9,stroke:#4caf50
    classDef operation fill:#e3f2fd,stroke:#1976d2
    classDef support fill:#fff3e0,stroke:#ff9800
    classDef management fill:#f3e5f5,stroke:#9c27b0
    classDef tech fill:#e0f7fa,stroke:#0097a7
    
    class A1,A2 customer
    class A3,A4,A5,A6 operation
    class A7,A8 support
    class A9,A10,A11,A14 management
    class A12,A13 tech

```
